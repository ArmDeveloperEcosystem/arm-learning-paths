{{/* Only returns a string, the properly formatted time in hrs and mins, preprocessed before site is served. Where it is used: - learning path list page - term list page - learning path introduction - install guide Called from: - partials learning-paths/introduction.html - learning-paths/list.html - _default/term.html - partials install-guides/installtoolsall.html */}} {{- $h := div (int .) 60 -}} {{- $m := mod (int .) 60 -}} {{- $formatted_time := "" -}} {{- if eq $h 0 -}} {{- $formatted_time = "" -}} {{- else if eq $h 1 -}} {{- $formatted_time = (print $h " hr") -}} {{- else if gt $h 1 -}} {{- $formatted_time = (print $h " hrs") -}} {{- end -}} {{- if eq $m 0 -}} {{- $formatted_time = $formatted_time -}} {{- else if and (ne $h 0) (ne $m 0) -}} {{- $formatted_time = (print $formatted_time " " $m " min") -}} {{- else -}} {{- $formatted_time = (print $m " min") -}} {{- end -}} {{$formatted_time}}
